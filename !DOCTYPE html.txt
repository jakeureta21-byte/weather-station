    <script>
        // REPLACE WITH YOUR FIREBASE CONFIG
        const firebaseConfig = {
            apiKey: "AIzaSyAj9oZblvzb7QLG6yC7VIzI0WQ8uqCkbio",
            databaseURL: "https://iot-weather-monitoring-s-b34e6-default-rtdb.firebaseio.com/"
        };
        
        firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        // 1. Live Clock
        setInterval(() => {
            const now = new Date();
            document.getElementById('date-time').innerText = now.toLocaleDateString() + " | " + now.toLocaleTimeString();
        }, 1000);

        // 2. Read Weather Data
        db.ref('Weather').on('value', (snap) => {
            const data = snap.val();
            document.getElementById('temp').innerText = data.Temperature || 0;
            document.getElementById('hum').innerText = data.Humidity || 0;
            document.getElementById('pres').innerText = data.Pressure || 0;
            document.getElementById('pm25').innerText = data.PM25 || 0;
            document.getElementById('wind').innerText = data.WindSpeed || 0;
            document.getElementById('dir').innerText = data.WindDirection || "N";
            document.getElementById('batt').innerText = data.BatteryVoltage || 0;
            
            const rainEl = document.getElementById('rain');
            if(data.Rain) { rainEl.innerText = "RAINING"; rainEl.classList.add('rain-alert'); }
            else { rainEl.innerText = "DRY"; rainEl.classList.remove('rain-alert'); }
        });

        // 3. Read Fan Status
        let currentFanState = 0;
        db.ref('Controls/FanStatus').on('value', (snap) => {
            currentFanState = snap.val();
            const fanEl = document.getElementById('fan-status');
            fanEl.innerText = currentFanState ? "ON" : "OFF";
            fanEl.className = currentFanState ? "value fan-on" : "value fan-off";
        });

        // 4. Manual Fan Toggle
        function toggleFan() {
            const newState = currentFanState === 1 ? 0 : 1;
            db.ref('Controls').update({ FanStatus: newState });
        }
    </script>
